138  basic_json(initializer_list const& init_list, json_type exact_type = json_type::null)
139  {
140  bool is_an_object = std::all_of(init_list.begin(), init_list.end(),
141  [](const basic_json& json)
142  { return (json.is_array() && json.size() == 2 && json[0].is_string()); });
143 
144  if (exact_type != json_type::object && exact_type != json_type::array)
145  {
146  exact_type = is_an_object ? json_type::object : json_type::array;
147  }
148 
149  if (exact_type == json_type::object)
150  {
151  JSONXX_ASSERT(is_an_object);
152  value_ = json_type::object;
153 
154  std::for_each(init_list.begin(), init_list.end(),
155  [this](const basic_json& json) {
156  value_.data.object->emplace(*((*json.value_.data.vector)[0].value_.data.string),
157  (*json.value_.data.vector)[1]);
158  });
159  }
160  else
161  {
162  value_ = json_type::array;
163  value_.data.vector->reserve(init_list.size());
164  value_.data.vector->assign(init_list.begin(), init_list.end());
165  }
166  }

374  template <class _IteratorTy, typename std::enable_if<std::is_same<_IteratorTy, iterator>::value
375  || std::is_same<_IteratorTy, const_iterator>::value,
376  int>::type = 0>
377  inline _IteratorTy erase(_IteratorTy pos)
378  {
379  _IteratorTy result = end();
380 
381  switch (type())
382  {
383  case json_type::object:
384  {
385  result.it_.object_iter = value_.data.object->erase(pos.it_.object_iter);
386  break;
387  }
388 
389  case json_type::array:
390  {
391  result.it_.array_iter = value_.data.vector->erase(pos.it_.array_iter);
392  break;
393  }
394 
395  default:
396  throw json_invalid_iterator("cannot use erase() with non-object & non-array value");
397  }
398 
399  return result;
400  }
